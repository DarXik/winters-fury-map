        private void Update()
        {
            if (fadeIn && myUIGroup.alpha < 1)
            {
                myUIGroup.alpha += Time.deltaTime;
                if (myUIGroup.alpha >= 1)
                {
                    fadeIn = false;
                }
            }

            for (int i = 0; i < buttons.Length && i < optionsObjects.Length; i++)
            {
                if (buttons[i] != null && optionsObjects[i] != null)
                {
                    buttons[i].interactable = !optionsObjects[i].activeSelf;
                }
                else
                {
                    Debug.Log("Nenalezeno");
                }
            }
